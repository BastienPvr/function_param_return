1 /**
  2  * Supprime les éléments d'une NodeList
  3  * @param   {NodeList} list La NodeList à analyser
  4  * @returns {Boolean} True si tout à été supprimé, false dans le cas contraire
  5  */
  6 function deleteNodes(list){
  7     var count=0;
  8     list.forEach(function(elem){
  9         elem.parentElement.removeChild(elem);
 10         count++
 11     });
 12     if(list.length===count){
 13         return true
 14     }else{
 15         return false
 16     }
 17 }

 19 /**
 20  * Vérifie si un nombre est pair
 21  * @throws {Error} Erreur si l'entrée n'est pas un nombre
 22  * @param   {Number} n Nombre à analyser
 23  * @returns {Boolean} True si n est pair, sinon false
 24  */
 25 function isOdd(n){
 26     if(typeof n !== "number"){
 27         throw new Error("Please pass a number as argument");
 28     }else if(n/2%1===0){
 29         return true
 30     }else{
 31         return false
 32     }
 33 }

 35 /**
 36  * Fait la somme de tous les nombres présents dans un tableau
 37  * @throws {Error} Erreur si une des entrés n'est pas un nombre
 38  * @param   {Array} a Tableau de nombres
 39  * @returns {Number} La somme des nombres présents dans le tableau
 40  */
 41 function sum(a){
 42     var sum = 0
 43     a.forEach(function(elem){
 44         if(typeof elem !== "number"){
 45             throw new Error("Please pass a number as argument");
 46         }else{
 47             sum+=elem
 48         }
 49     });
 50     return sum
 51 }

 53 /**
 54  * Récupère la largeur et la hauteur de tous les éléments présents dans une NodeList
 55  * @param   {NodeList} nl La NodeList à analyser
 56  * @returns {Array} Tableau d'objets
 57  */
 58 function getNodesMetrics(nl){
 59     var table=[];
 60     var count = 0
 61     nl.forEach(function(elem){
 62         var newObject = {};
 63         newObject.tabindex = count;
 64         newObject.node = elem;
 65         newObject.width = elem.getBoundingClientRect().width;
 66         newObject.height = elem.getBoundingClientRect().height;
 67         table.push(newObject);
 68         count++;
 69     });
 70     return table
 71 }
